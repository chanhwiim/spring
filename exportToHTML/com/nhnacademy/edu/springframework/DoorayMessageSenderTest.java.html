package com.nhnacademy.edu.springframework;

import com.nhnacademy.edu.springframework.domain.User;
import com.nhnacademy.edu.springframework.service.impl.DoorayMessageSender;
import org.junit.jupiter.api.Assertions;
import org.junit.jupiter.api.Test;


public class DoorayMessageSenderTest {

    @Test
    public void testSendMessageSuccess() {
        DoorayMessageSender doorayMessageSender = new DoorayMessageSender();
        User user = new User("email", "phoneNumber");
        String message = "test message";

        Assertions.assertDoesNotThrow(()-> doorayMessageSender.sendMessage(user, message));
    }

    @Test
    public void testSendMessageFail() {
        DoorayMessageSender doorayMessageSender = new DoorayMessageSender();
        User user = new User("email", "phoneNumber");
        String message = null;

        Assertions.assertDoesNotThrow(()-> doorayMessageSender.sendMessage(user, message));
    }
}
